const DEBT_MULTIPLIER=0.30,HH_CONNECTION_COST=425,HH_FRACTION=0.999,

  NUMA_FRACTION=0.001,NUMA_SYSTEM_COST=52000,TARIFF=1.40,HH_MONTHLY_VOL=4.4,

  RETAIL_MONTHLY_VOL=40.0,RETAINED_PCT=0.20,RETENTION_GROWTH=1.02,

  PEOPLE_PER_HH=9.7,CUSTOMER_DEPOSIT_PER_HH=80;


function calculate(annualDonation,donationPeriod,horizon){

  const years=Math.max(horizon,donationPeriod);

  const donation=[],debtUnlocked=[],customerDeposit=[],capex=[],hhPerYear=[];

  for(let y=0;y<years;y++){

    const don=y<donationPeriod?annualDonation:0;

    const debt=don*DEBT_MULTIPLIER;

    const hhDirect=don>0?(don*HH_FRACTION+debt)/HH_CONNECTION_COST:0;

    const custDep=hhDirect*CUSTOMER_DEPOSIT_PER_HH;

    const hhCust=custDep/HH_CONNECTION_COST;

    donation.push(don);debtUnlocked.push(debt);

    customerDeposit.push(custDep);capex.push(don+debt+custDep);

    hhPerYear.push(hhDirect+hhCust);

  }

  const annualSales=[];let cumHH=0;

  for(let y=0;y<years;y++){

    cumHH+=hhPerYear[y];

    const cumCapex=capex.slice(0,y+1).reduce((a,b)=>a+b,0);

    const numaFunded=(cumCapex*NUMA_FRACTION)/NUMA_SYSTEM_COST;

    const hhRev=cumHH*HH_MONTHLY_VOL*TARIFF*12;

    const retailRev=numaFunded*3*RETAIL_MONTHLY_VOL*0.70*TARIFF*12;

    annualSales.push(hhRev+retailRev);

  }

  const annualRetained=[];let baseRetained=0;

  for(let y=0;y<years;y++){

    let retained;

    if(y<donationPeriod){retained=annualSales[y]*RETAINED_PCT;baseRetained=retained;}

    else{retained=baseRetained*Math.pow(RETENTION_GROWTH,y-donationPeriod+1);}

    annualRetained.push(retained);

  }

  let cumRetainedAcc=0;

  const cumRetained=annualRetained.map(r=>{cumRetainedAcc+=r;return cumRetainedAcc;});

  const hhFromRetained=annualRetained.map((r,y)=>(r*(years-y))/HH_CONNECTION_COST);

  const totalDonations=donation.reduce((a,b)=>a+b,0);

  const totalDebtCustomer=debtUnlocked.reduce((a,b)=>a+b,0)+customerDeposit.reduce((a,b)=>a+b,0);

  const totalCapex=capex.reduce((a,b)=>a+b,0);

  const totalCumRetained=cumRetained[years-1];

  const cumulativeImpact=totalCapex+totalCumRetained;

  const totalHH=hhPerYear.reduce((a,b)=>a+b,0)+hhFromRetained.reduce((a,b)=>a+b,0);

  const totalPeople=totalHH*PEOPLE_PER_HH;

  const multiplier=totalDonations>0?cumulativeImpact/totalDonations:0;

  const rows=[];let cumDon=0,cumImpact=0;

  for(let y=0;y<years;y++){

    cumDon+=donation[y];

    const annualImpact=capex[y]+annualRetained[y];

    cumImpact+=annualImpact;

    rows.push({year:y+1,donation:donation[y],debtCust:debtUnlocked[y]+customerDeposit[y],

      annualRetained:annualRetained[y],annualImpact,cumDonations:cumDon,cumImpact,

      multiplier:cumDon>0?cumImpact/cumDon:0,isDonationYear:y<donationPeriod});

  }

  return{totalDonations,totalDebtCustomer,totalCapex,totalCumRetained,

    cumulativeImpact,multiplier,totalHH:Math.round(totalHH),

    totalPeople:Math.round(totalPeople),rows,years};

}

const fmtX=n=>n.toLocaleString('en-US',{minimumFractionDigits:2,maximumFractionDigits:2})+'×';

const fmtN=n=>n.toLocaleString('en-US',{maximumFractionDigits:0});


function animateCount(el,target){

  const start=performance.now(),duration=600;

  const startVal=parseInt(el.dataset.val||'0');

  el.dataset.val=target;

  function step(now){

    const p=Math.min((now-start)/duration,1);

    const ease=1-Math.pow(1-p,3);

    const cur=Math.round(startVal+(target-startVal)*ease);

    el.textContent=fmtN(cur);

    if(p<1)requestAnimationFrame(step);

    else el.textContent=fmtN(target);

  }

  requestAnimationFrame(step);

}

function renderChart(r){

  const ctx=document.getElementById('impact-chart').getContext('2d');

  if(window._chart){window._chart.destroy();window._chart=null;}

  const dark=document.documentElement.getAttribute('data-theme')==='dark';

  const pc=dark?'#38A8CC':'#006D8F',ac=dark?'#00C4DD':'#00B0C8';

  const tm=dark?'#7BA0B8':'#4E6070',gc=dark?'rgba(255,255,255,.06)':'rgba(10,37,64,.07)';

  const sc=dark?'#111F2C':'#FFF',tc=dark?'#D4E5F0':'#0A2540';

  window._chart=new Chart(ctx,{

    type:'line',

    data:{labels:r.rows.map(row=>`Yr ${row.year}`),

      datasets:[

        {label:'Cumulative Donations',data:r.rows.map(row=>row.cumDonations),

          borderColor:pc,backgroundColor:pc+'18',borderWidth:2.5,pointRadius:4,

          pointBackgroundColor:pc,pointBorderColor:sc,pointBorderWidth:2,fill:false,tension:.3},

        {label:'Cumulative Impact',data:r.rows.map(row=>row.cumImpact),

          borderColor:ac,backgroundColor:ac+'18',borderWidth:2.5,pointRadius:4,

          pointBackgroundColor:ac,pointBorderColor:sc,pointBorderWidth:2,fill:'+1',tension:.3}

      ]},

    options:{responsive:true,maintainAspectRatio:false,

      interaction:{mode:'index',intersect:false},

      plugins:{legend:{display:false},

        tooltip:{backgroundColor:tc+'ee',titleColor:sc,bodyColor:sc+'cc',padding:12,cornerRadius:8,

          callbacks:{label:item=>' '+item.dataset.label+': '+fmt(item.raw)}}},

      scales:{

        x:{grid:{color:gc},ticks:{color:tm,font:{family:'DM Sans,sans-serif',size:12}}},

        y:{grid:{color:gc},ticks:{color:tm,font:{family:'DM Sans,sans-serif',size:12},

          callback:v=>'$'+(v>=1000000?(v/1000000).toFixed(1)+'M':v>=1000?(v/1000).toFixed(0)+'K':v)}}

      }}

  });

}

let compareOn=false;

  compareOn=!compareOn;

function renderCompareResult(el,r,otherR,isB){

  const better=isB?r.multiplier>otherR.multiplier:r.multiplier>=otherR.multiplier;

  el.innerHTML=`

    <div class="compare-result-row"><span class="compare-result-lbl">People served</span>

      <span class="compare-result-val ${compareOn&&better?'better':''}">${fmtN(r.totalPeople)}</span></div>

    <div class="compare-result-row"><span class="compare-result-lbl">Cumulative impact</span>

      <span class="compare-result-val ${compareOn&&better?'better':''}">${fmt(r.cumulativeImpact)}</span></div>

    <div class="compare-result-row"><span class="compare-result-lbl">Multiplier</span>

      <span class="compare-result-val ${compareOn&&better?'better':''}">${fmtX(r.multiplier)}</span></div>

    <div class="compare-result-row"><span class="compare-result-lbl">Total donated</span>

      <span class="compare-result-val">${fmt(r.totalDonations)}</span></div>`;

}

function recalcCompare(){

  const horizon=parseInt(document.getElementById('impact-horizon').value)||10;

  const aAmt=parseFloat(document.getElementById('cmp-a-amt').value)||167000;

  const aPeriod=parseInt(document.getElementById('cmp-a-period').value)||3;

  const bAmt=parseFloat(document.getElementById('cmp-b-amt').value)||50000;

  const bPeriod=parseInt(document.getElementById('cmp-b-period').value)||5;

  const rA=calculate(aAmt,aPeriod,Math.max(horizon,aPeriod));

  const rB=calculate(bAmt,bPeriod,Math.max(horizon,bPeriod));

  renderCompareResult(document.getElementById('cmp-a-result'),rA,rB,false);

  renderCompareResult(document.getElementById('cmp-b-result'),rB,rA,true);

}

function recalc(){

  const donation=parseFloat(document.getElementById('annual-donation').value)||0;

  const period=parseInt(document.getElementById('donation-period').value)||3;

  const horizon=parseInt(document.getElementById('impact-horizon').value)||10;

  if(!donation||donation<100)return;

  const r=calculate(donation,period,Math.max(horizon,period));

  document.getElementById('empty-state').style.display='none';

  const el=document.getElementById('results');

  el.classList.add('visible');

  renderResults(r);

  updateNamedGift(donation);

  document.getElementById('cmp-a-amt').value=donation;

  document.getElementById('cmp-a-period').value=period;

}


window.addEventListener('DOMContentLoaded',()=>{

  });
